// reviewCmd executes a PR review by orchestrating all enabled consuls.
var reviewCmd = &cobra.Command{
        Use:   "review",
        Short: "Review a GitHub PR using multiple AI agents",
        Long: `Fetches the PR diff and metadata from GitHub via the gh CLI, then fans
out the review to all enabled AI consuls in parallel. After all consuls finish,
runs a final Claude aggregation to produce a consolidated review.`,
        RunE: runReview,
}

func init() {
        reviewCmd.Flags().StringVar(&prNumber, "pr", "", "GitHub PR number to review (required)")
        _ = reviewCmd.MarkFlagRequired("pr")
        rootCmd.AddCommand(reviewCmd)
}

// runReview is the handler for the review subcommand.
func runReview(cmd *cobra.Command, _ []string) error {
        // 1. Resolve config path.
        configPath := cfgFile
        if configPath == "" {
                home, err := os.UserHomeDir()
                if err != nil {
                        return fmt.Errorf("resolving home dir: %w", err)
                }
                configPath = filepath.Join(home, ".consul-review", "config.yaml")
        }

        // 2. Load config.
        cfg, err := config.Load(configPath)
        if err != nil {
                return fmt.Errorf("loading config: %w", err)
        }

        // 3. Structural validation.
        if err := config.Validate(cfg); err != nil {
                return fmt.Errorf("invalid config: %w", err)
        }

        // 4. Binary pre-flight check — exit immediately with clear error if missing.
        if err := config.CheckBinaries(cfg); err != nil {
                return fmt.Errorf("pre-flight check failed: %w", err)
        }

        // 5. Resolve skill content: use the user-specified file OR fall back to the
        //    bundled go-code-review skill embedded in the binary.
        var skillContent string
        if cfg.CodeReviewSkill != "" {
                b, err := os.ReadFile(cfg.CodeReviewSkill)
                if err != nil {
                        return fmt.Errorf("reading skill file %q: %w", cfg.CodeReviewSkill, err)
                }
                skillContent = string(b)
                slog.Info("using custom skill file", "path", cfg.CodeReviewSkill)
        } else {
                skillContent = assets.DefaultCodeReviewSkill
                slog.Info("using bundled default skill (go-code-review)")
        }

        slog.Info("starting consul-review", "pr", prNumber, "config", configPath)

        // 6. Orchestrate.
        r := runner.OSRunner{}
        if err := orchestrator.Orchestrate(cmd.Context(), cfg, skillContent, prNumber, r); err != nil {
                return fmt.Errorf("review failed: %w", err)
        }

        slog.Info("consul-review complete", "pr", prNumber)
        return nil
}

                        versionInfo.version, versionInfo.commit, versionInfo.date, versionInfo.builtBy,

        }

// Package config loads and validates consul-review configuration.
package config

import (
        "errors"
        "fmt"
        "os"
        "os/exec"
        "path/filepath"
        "strings"

        "github.com/spf13/viper"
)

// SupportedConsuls maps consul name → binary name.
var SupportedConsuls = map[string]string{
        "gemini":  "gemini",
        "copilot": "copilot",
        "oz":      "oz",
}

// consulEnvVars maps consul name → environment variable for its API key.
var consulEnvVars = map[string]string{
        "gemini":  "GEMINI_API_KEY",
        "copilot": "GH_TOKEN",
        "oz":      "OZ_API_KEY",
}

// EnvVarFor returns the environment variable name for the given consul.
// Returns an empty string for unknown consul names.
func EnvVarFor(consulName string) string {
        return consulEnvVars[consulName]
}

// ConsulConfig holds common per-consul configuration. Each consul is a named
// top-level field in Config so that consul-specific fields can be added to
// specialised structs (e.g. OzConfig) without breaking others.
type ConsulConfig struct {
        Enabled bool   `mapstructure:"enabled"`
        Model   string `mapstructure:"model"`
        // APIKey is the literal API key value; the orchestrator injects it into the
        // appropriate environment variable via os.Setenv before spawning the process.
        APIKey string `mapstructure:"api_key"`
}

// Config is the root configuration structure.
// Each consul is a named top-level field, which allows adding consul-specific
// configuration in the future without a map lookup and without breaking other
// consul configs.
type Config struct {
        // CodeReviewSkill is optional. When empty the binary's bundled skill is used.
        CodeReviewSkill string       `mapstructure:"code_review_skill"`
        Gemini          ConsulConfig `mapstructure:"gemini"`
        Copilot         ConsulConfig `mapstructure:"copilot"`
        Oz              ConsulConfig `mapstructure:"oz"`
}

// EnabledConsuls returns a map of consul name → ConsulConfig for every consul
// that has Enabled = true. This gives the orchestrator a stable iteration
// interface regardless of the concrete per-consul field layout.
func (c *Config) EnabledConsuls() map[string]ConsulConfig {
        all := map[string]ConsulConfig{
                "gemini":  c.Gemini,
                "copilot": c.Copilot,
                "oz":      c.Oz,
        }
        enabled := make(map[string]ConsulConfig, len(all))
        for name, cc := range all {
                if cc.Enabled {
                        enabled[name] = cc
                }
        }
        return enabled
}

// Load reads the YAML config at path (expanding ~ first) and unmarshals it.
func Load(path string) (*Config, error) {
        expanded, err := expandTilde(path)
        if err != nil {
                return nil, fmt.Errorf("expanding config path: %w", err)
        }

        v := viper.New()
        v.SetConfigFile(expanded)
        v.SetConfigType("yaml")

        if err := v.ReadInConfig(); err != nil {
                return nil, fmt.Errorf("reading config %q: %w", expanded, err)
        }

        var cfg Config
        if err := v.Unmarshal(&cfg); err != nil {
                return nil, fmt.Errorf("unmarshaling config: %w", err)
        }

        cfg.CodeReviewSkill, err = expandTilde(cfg.CodeReviewSkill)
        if err != nil {
                return nil, fmt.Errorf("expanding code_review_skill path: %w", err)
        }

        return &cfg, nil
}

// Validate performs structural validation of the config.
//   - code_review_skill, if set, must exist on disk (empty = use bundled default)
//   - At least one consul must be enabled
func Validate(c *Config) error {
        var errs []string

        // code_review_skill is optional — empty means use the bundled default.
        if c.CodeReviewSkill != "" {
                if _, err := os.Stat(c.CodeReviewSkill); err != nil {
                        errs = append(errs, fmt.Sprintf("code_review_skill %q not found: %v", c.CodeReviewSkill, err))
                }
        }

        if len(c.EnabledConsuls()) == 0 {
                errs = append(errs, "at least one consul must be enabled")
        }

        if len(errs) > 0 {
                return errors.New(strings.Join(errs, "; "))
        }
        return nil
}

// LookPathFunc is the function used to locate binaries.
// It is a package-level variable so tests can inject a fake.
var LookPathFunc = exec.LookPath

// CheckBinaries verifies that every enabled consul's binary and gh are present
// in PATH. Returns a single error listing all missing binaries.
func CheckBinaries(c *Config) error {
        required := []string{"gh"}

        for name := range c.EnabledConsuls() {
                required = append(required, SupportedConsuls[name])
        }

        var missing []string
        for _, bin := range required {
                if _, err := LookPathFunc(bin); err != nil {
                        missing = append(missing, bin)
                }
        }

        if len(missing) > 0 {
                return fmt.Errorf("required binaries not found in PATH: %s", strings.Join(missing, ", "))
        }
        return nil
}

// expandTilde replaces a leading ~ with the user's home directory.
func expandTilde(path string) (string, error) {
        if !strings.HasPrefix(path, "~") {
                return path, nil
        }
        home, err := os.UserHomeDir()
        if err != nil {
                return "", fmt.Errorf("resolving home directory: %w", err)
        }
        return filepath.Join(home, path[1:]), nil
}

package orchestrator

import (

// consulArgs describes the binary and argument list for a consul invocation.
type consulArgs struct {
        bin  string
        args []string
}

// buildArgs constructs the binary and CLI arguments for the named consul.
// All commands are forced into non-interactive mode via their respective flags.
// Returns an error for unknown consul names.
func buildArgs(consulName, model, prompt string) (consulArgs, error) {
        switch consulName {
        case "gemini":
                return consulArgs{
                        bin:  "gemini",
                        args: []string{"-p", prompt, "--yolo", "--model", model},
                }, nil
        case "copilot":
                return consulArgs{
                        bin:  "copilot",
                        args: []string{"-p", prompt, "--allow-all-tools"},
                }, nil
        case "oz":
                return consulArgs{
                        bin:  "oz",
                        args: []string{"agent", "run", "--prompt", prompt, "--no-interactive"},
                }, nil
        default:
                return consulArgs{}, fmt.Errorf("unknown consul %q", consulName)
        }
}

// buildPrompt combines the skill file content and PR metadata into a single
// prompt string that is passed verbatim to each consul agent.
func buildPrompt(skillContent, prTitle, prBody, diff string) string {
        var sb strings.Builder
        sb.WriteString("## Code Review Skill\n\n")
        sb.WriteString(skillContent)
        sb.WriteString("\n\n## Pull Request\n\n")
        sb.WriteString(fmt.Sprintf("**Title**: %s\n\n", prTitle))
        if prBody != "" {
                sb.WriteString(fmt.Sprintf("**Description**:\n%s\n\n", prBody))
        }
        sb.WriteString("## Diff\n\n```diff\n")
        sb.WriteString(diff)
        sb.WriteString("\n```\n")
        return sb.String()
}

// injectEnv sets the appropriate environment variable for the consul's API key
// before the subprocess is started. The variable is set at the process level
// so that child processes inherit it.
func injectEnv(consulName, apiKey string) {
        envVar := config.EnvVarFor(consulName)
        if envVar == "" || apiKey == "" {
                return
        }
        if err := os.Setenv(envVar, apiKey); err != nil {
                slog.Warn("failed to set env var", "var", envVar, "consul", consulName, "err", err)
        }
}

// PRMeta holds the data fetched from the GitHub API via gh CLI.
type PRMeta struct {
        Title string
        Body  string
        Diff  string
}

// fetchPR retrieves the PR title, body, and unified diff for prNumber using the
// gh CLI. Both calls are made through the Runner interface so they are mockable.
func fetchPR(ctx context.Context, r runner.Runner, prNumber string) (PRMeta, error) {
        // 1. Fetch PR metadata (title + body) as JSON.
        var metaBuf bytes.Buffer
        if err := r.Run(ctx, "gh", []string{
                "pr", "view", prNumber,
                "--json", "title,body",
                "--jq", `"TITLE:\(.title)\nBODY:\(.body)"`,
        }, nil, &metaBuf); err != nil {
                return PRMeta{}, fmt.Errorf("gh pr view: %w", err)
        }

        // 2. Fetch unified diff.
        var diffBuf bytes.Buffer
        if err := r.Run(ctx, "gh", []string{"pr", "diff", prNumber}, nil, &diffBuf); err != nil {
                return PRMeta{}, fmt.Errorf("gh pr diff: %w", err)
        }

        meta := parsePRMeta(metaBuf.String())
        meta.Diff = diffBuf.String()
        return meta, nil
}

// parsePRMeta extracts title and body from the simple TITLE:/BODY: format
// produced by the gh jq query.
func parsePRMeta(raw string) PRMeta {
        var m PRMeta
        for _, line := range strings.SplitN(raw, "\n", 3) {
                switch {
                case strings.HasPrefix(line, "TITLE:"):
                        m.Title = strings.TrimPrefix(line, "TITLE:")
                case strings.HasPrefix(line, "BODY:"):
                        m.Body = strings.TrimPrefix(line, "BODY:")
                }
        }
        return m
}

// workerResult holds the output path and any error from a single consul run.
type workerResult struct {
        consulName string
        outputFile string
        err        error
}

// Orchestrate is the main entry point. It:
//  1. Fetches PR metadata and diff via gh CLI.
//  2. Fans out to all enabled consuls in parallel goroutines.
//  3. Collects output files and errors.
//  4. Runs a final Claude aggregation across the output files.
//
// Concurrency: each consul runs in its own goroutine controlled by a WaitGroup.
// Errors are collected into a slice guarded by a Mutex so no goroutine is blocked.
func Orchestrate(ctx context.Context, cfg *config.Config, skillContent, prNumber string, r runner.Runner) error {
        slog.Info("fetching PR", "pr", prNumber)
        pr, err := fetchPR(ctx, r, prNumber)
        if err != nil {
                return fmt.Errorf("fetching PR #%s: %w", prNumber, err)
        }

        prompt := buildPrompt(skillContent, pr.Title, pr.Body, pr.Diff)

        outDir, err := os.MkdirTemp("", "consul-review-*")
        if err != nil {
                return fmt.Errorf("creating temp dir: %w", err)
        }
        slog.Info("review outputs will be written to", "dir", outDir)

        enabled := cfg.EnabledConsuls()
        results := make([]workerResult, 0, len(enabled))
        var (
                wg sync.WaitGroup
                mu sync.Mutex
        )

        for name, cc := range enabled {
                // Capture loop variables for closure safety.
                consulName := name
                consulCfg := cc

                wg.Add(1)
                go func() {
                        defer wg.Done()

                        res := runConsul(ctx, r, consulName, consulCfg, prompt, outDir)

                        mu.Lock()
                        results = append(results, res)
                        mu.Unlock()
                }()
        }

        wg.Wait()

        // Collect output files and surface errors.
        var outputFiles []string
        var runErrs []string
        for _, res := range results {
                if res.err != nil {
                        slog.Error("consul failed", "consul", res.consulName, "err", res.err)
                        runErrs = append(runErrs, fmt.Sprintf("%s: %v", res.consulName, res.err))
                } else {
                        outputFiles = append(outputFiles, res.outputFile)
                }
        }

        if len(outputFiles) == 0 {
                return fmt.Errorf("all consuls failed: %s", strings.Join(runErrs, "; "))
        }

        if err := runAggregation(ctx, r, outputFiles); err != nil {
                return fmt.Errorf("aggregation step: %w", err)
        }

        if len(runErrs) > 0 {
                return fmt.Errorf("partial failure — some consuls errored: %s", strings.Join(runErrs, "; "))
        }
        return nil
}

// runConsul injects the API key env var, builds the command args, runs the
// consul binary, and writes its stdout to a review file in outDir.
func runConsul(ctx context.Context, r runner.Runner, name string, cc config.ConsulConfig, prompt, outDir string) workerResult {
        // Inject API key into process environment before spawning subprocess.
        injectEnv(name, cc.APIKey)

        ca, err := buildArgs(name, cc.Model, prompt)
        if err != nil {
                return workerResult{consulName: name, err: err}
        }

        outPath := filepath.Join(outDir, fmt.Sprintf("review_%s.md", name))
        f, err := os.Create(outPath)
        if err != nil {
                return workerResult{consulName: name, err: fmt.Errorf("creating output file: %w", err)}
        }
        defer f.Close()

        slog.Info("starting consul", "consul", name, "binary", ca.bin)
        if err := r.Run(ctx, ca.bin, ca.args, nil, f); err != nil {
                return workerResult{consulName: name, err: fmt.Errorf("running %s: %w", ca.bin, err)}
        }

        slog.Info("consul finished", "consul", name, "output", outPath)
        return workerResult{consulName: name, outputFile: outPath}
}

// runAggregation calls claude with all review output files to produce a
// consolidated final review printed to stdout.
func runAggregation(ctx context.Context, r runner.Runner, outputFiles []string) error {
        prompt := fmt.Sprintf(
                "Analyze the following AI code review outputs and produce a single, concise, consolidated review summary highlighting the most critical findings, areas of agreement, and any conflicting opinions: %s",
                strings.Join(outputFiles, " "),
        )

        slog.Info("running aggregation", "files", outputFiles)
        return r.Run(ctx, "claude", []string{prompt}, nil, os.Stdout)
}

package runner

import (
        "context"
        "fmt"
        "io"
        "sync"
)

// Call records a single invocation made through FakeRunner.
type Call struct {
        Name string
        Args []string
}

// FakeRunner is a deterministic test double for Runner.
// It records every call and writes a configurable Output to stdout.
// If Err is non-nil it is returned from Run (after writing Output).
type FakeRunner struct {
        mu     sync.Mutex
        Calls  []Call
        Output string
        Err    error
}

// Run records the call, writes Output to stdout, then returns Err.
func (f *FakeRunner) Run(_ context.Context, name string, args []string, _ io.Reader, stdout io.Writer) error {
        f.mu.Lock()
        f.Calls = append(f.Calls, Call{Name: name, Args: args})
        f.mu.Unlock()

        if f.Output != "" {
                if _, err := fmt.Fprint(stdout, f.Output); err != nil {
                        return err
                }
        }
        return f.Err
}

// CallCount returns the number of times Run was invoked. Safe for concurrent use.
func (f *FakeRunner) CallCount() int {
        f.mu.Lock()
        defer f.mu.Unlock()
        return len(f.Calls)
}

// CalledWith returns true if the runner was called with the given binary name.
func (f *FakeRunner) CalledWith(name string) bool {
        f.mu.Lock()
        defer f.mu.Unlock()
        for _, c := range f.Calls {
                if c.Name == name {
                        return true
                }
        }
        return false
}

// Package runner provides an abstraction for executing external commands.
package runner

import (
        "context"
        "io"
        "os/exec"
)

// Runner executes an external binary and streams its stdout to the provided writer.
// Implementing this interface for each production binary or a fake in tests keeps
// subprocess logic fully decoupled from orchestration logic.
type Runner interface {
        Run(ctx context.Context, name string, args []string, stdin io.Reader, stdout io.Writer) error
}

// OSRunner is the production Runner that shells out via os/exec.
type OSRunner struct{}

// Run executes the binary at name with args, wiring stdin and stdout. Stderr is
// discarded; callers should redirect it if desired. The command is cancelled when
// ctx is done.
func (OSRunner) Run(ctx context.Context, name string, args []string, stdin io.Reader, stdout io.Writer) error {
        cmd := exec.CommandContext(ctx, name, args...)
        cmd.Stdin = stdin
        cmd.Stdout = stdout
        return cmd.Run()
}
